<script>
/* ── State ── */
let currentLang = 'ar';
let currentMethod = { login:'email', reg:'email', forgot:'email' };
let timerInterval = null;
let timerSecs = 60;
let otpResendAllowed = false;
let userInfo = { name:'', contact:'' };

/* ── Page navigation ── */
function goToPage(id){
  document.querySelectorAll('.page').forEach(p=>p.classList.remove('active'));
  document.getElementById(id).classList.add('active');
  window.scrollTo(0,0);
  const authBar = document.getElementById('authBar');
  authBar.style.display = id === 'page-dash' ? 'none' : 'flex';
}

/* ── Lang ── */
function setLang(lang){
  currentLang = lang;
  document.documentElement.setAttribute('data-lang', lang);
  document.documentElement.setAttribute('dir', lang==='ar'?'rtl':'ltr');
  document.documentElement.setAttribute('lang', lang);
  ['btn-ar','btn-en','dash-ar','dash-en'].forEach(id=>{
    const el=document.getElementById(id);
    if(el){el.classList.toggle('on', el.id.endsWith(lang==='ar'?'ar':'en'))}
  });
  updateTimerDisplay();
}

/* ── Method toggle ── */
function switchMethod(method, form){
  currentMethod[form] = method;
  const prefix = form === 'login' ? 'login' : form === 'reg' ? 'reg' : 'forgot';
  const emailEl = document.getElementById(prefix + (form==='login'?'-email':'reg'==='reg'?'-email-field':'forgot'==='forgot'?'-email-field':'-email'));
  // Simpler direct IDs
  const ids = {
    login: { email:'login-email', phone:'login-phone' },
    reg:   { email:'reg-email-field', phone:'reg-phone-field' },
    forgot:{ email:'forgot-email-field', phone:'forgot-phone-field' }
  };
  document.getElementById(ids[form].email).style.display = method==='email'?'block':'none';
  document.getElementById(ids[form].phone).style.display = method==='phone'?'block':'none';

  // Update toggle buttons
  const btns = document.querySelectorAll(`#page-${form==='login'?'login':form==='reg'?'register':'forgot'} .mtoggle`);
  btns.forEach((b,i)=>b.classList.toggle('active', (i===0&&method==='email')||(i===1&&method==='phone')));
}

/* ── Password strength ── */
function checkPwStrength(input){
  const pw = input.value;
  const str = document.getElementById('pw-str');
  const lbl = document.getElementById('pw-lbl');
  str.classList.toggle('show', pw.length > 0);
  if(!pw.length) return;
  let score = 0;
  if(pw.length >= 6) score++;
  if(pw.length >= 10) score++;
  if(/[A-Z]/.test(pw) && /[a-z]/.test(pw)) score++;
  if(/[0-9]/.test(pw) && /[^a-zA-Z0-9]/.test(pw)) score++;
  const bars = [document.getElementById('pb1'),document.getElementById('pb2'),document.getElementById('pb3'),document.getElementById('pb4')];
  const cls = score<=1?'weak':score<=2?'mid':'strong';
  const labels = { ar: {weak:'ضعيفة',mid:'متوسطة',strong:'قوية'}, en: {weak:'Weak',mid:'Medium',strong:'Strong'} };
  bars.forEach((b,i)=>{ b.className='pw-bar '+(i<score?cls:'') });
  lbl.textContent = labels[currentLang][cls];
  lbl.style.color = cls==='weak'?'var(--danger)':cls==='mid'?'var(--warn)':'var(--ac3)';
}

/* ── Login ── */
function handleLogin(){
  const method = currentMethod.login;
  const contact = method==='email'
    ? document.getElementById('login-email-input').value.trim()
    : document.getElementById('login-phone-input').value.trim();
  const pw = document.getElementById('login-pw').value;
  if(!contact){ showToast(currentLang==='ar'?'أدخل بريدك الإلكتروني أو رقم جوالك':'Enter email or phone','error'); return; }
  if(method==='email' && pw.length < 1){ showToast(currentLang==='ar'?'أدخل كلمة المرور':'Enter password','error'); return; }
  setLoading('login-loader','login-btn-txt',true);
  userInfo = { name: contact.includes('@')?contact.split('@')[0]:'مستخدم', contact };
  setTimeout(()=>{
    setLoading('login-loader','login-btn-txt',false);
    launchOTP(contact, method);
  }, 1600);
}

/* ── Register ── */
function handleRegister(){
  const name = document.getElementById('reg-name').value.trim();
  const method = currentMethod.reg;
  const contact = method==='email'
    ? document.getElementById('reg-email').value.trim()
    : document.getElementById('reg-phone').value.trim();
  const pw = document.getElementById('reg-pw').value;
  const terms = document.getElementById('terms').checked;
  if(!name){ showToast(currentLang==='ar'?'أدخل اسمك الكامل':'Enter your full name','error'); return; }
  if(!contact){ showToast(currentLang==='ar'?'أدخل بريدك أو جوالك':'Enter email or phone','error'); return; }
  if(pw.length < 6){ showToast(currentLang==='ar'?'كلمة المرور قصيرة جداً':'Password too short','error'); return; }
  if(!terms){ showToast(currentLang==='ar'?'يجب الموافقة على الشروط':'Accept terms first','error'); return; }
  setLoading('reg-loader','reg-btn-txt',true);
  userInfo = { name, contact };
  setTimeout(()=>{
    setLoading('reg-loader','reg-btn-txt',false);
    launchOTP(contact, method);
  }, 1600);
}

/* ── Forgot ── */
function handleForgot(){
  showToast(currentLang==='ar'?'✅ تم إرسال رابط الاستعادة':'✅ Reset code sent!','success');
  setTimeout(()=>goToPage('page-login'), 2000);
}

/* ── Launch OTP ── */
function launchOTP(contact, method){
  document.getElementById('otp-destination').textContent = contact;
  document.getElementById('otp-normal-view').style.display='flex';
  document.getElementById('otp-success-view').style.display='none';
  // Clear boxes
  document.querySelectorAll('.otp-box').forEach(b=>{ b.value=''; b.classList.remove('filled','success','error-box'); });
  goToPage('page-otp');
  startTimer();
  showToast(
    currentLang==='ar'
      ? `📱 تم إرسال كود التحقق إلى ${contact}`
      : `📱 Verification code sent to ${contact}`,
    'success'
  );
  // Focus first box
  setTimeout(()=>document.querySelectorAll('.otp-box')[0].focus(), 300);
}

/* ── OTP navigation ── */
function otpFill(input){
  input.value = input.value.replace(/[^0-9]/g,'').slice(-1);
  input.classList.toggle('filled', input.value !== '');
}
function otpNav(input, idx){
  const boxes = document.querySelectorAll('.otp-box');
  if(event.key === 'Backspace' && !input.value && idx > 0){
    boxes[idx-1].focus();
    boxes[idx-1].value='';
    boxes[idx-1].classList.remove('filled');
  } else if(input.value && idx < 5){
    boxes[idx+1].focus();
  }
  // Auto verify
  const full = [...boxes].map(b=>b.value).join('');
  if(full.length === 6) setTimeout(()=>verifyOTP(), 200);
}

/* ── Verify OTP ── */
function verifyOTP(){
  const boxes = document.querySelectorAll('.otp-box');
  const code = [...boxes].map(b=>b.value).join('');
  if(code.length < 6){ showToast(currentLang==='ar'?'أدخل الكود كاملاً':'Enter full code','error'); return; }
  document.getElementById('otp-loader').style.display='block';
  setTimeout(()=>{
    document.getElementById('otp-loader').style.display='none';
    if(code === '123456'){
      boxes.forEach(b=>b.classList.add('success'));
      clearInterval(timerInterval);
      showToast(currentLang==='ar'?'✅ تم التحقق بنجاح!':'✅ Verified successfully!','success');
      document.getElementById('otp-normal-view').style.display='none';
      document.getElementById('otp-success-view').style.display='flex';
      setTimeout(()=>enterDashboard(), 2000);
    } else {
      boxes.forEach(b=>b.classList.add('error-box'));
      setTimeout(()=>boxes.forEach(b=>{ b.classList.remove('error-box'); b.value=''; b.classList.remove('filled'); }), 700);
      boxes[0].focus();
      showToast(currentLang==='ar'?'❌ الكود غير صحيح — جرب 123456':'❌ Wrong code — try 123456','error');
    }
  }, 900);
}

/* ── Resend OTP ── */
function resendOTP(){
  if(!otpResendAllowed) return;
  otpResendAllowed = false;
  document.getElementById('resend-btn').classList.remove('active');
  showToast(currentLang==='ar'?'📱 تم إعادة إرسال الكود':'📱 Code resent','success');
  timerSecs = 60;
  startTimer();
  document.querySelectorAll('.otp-box').forEach(b=>{ b.value=''; b.classList.remove('filled','success','error-box'); });
  document.querySelectorAll('.otp-box')[0].focus();
}

/* ── Timer ── */
function startTimer(){
  clearInterval(timerInterval);
  timerSecs = 60;
  otpResendAllowed = false;
  document.getElementById('resend-btn').classList.remove('active');
  updateTimerDisplay();
  timerInterval = setInterval(()=>{
    timerSecs--;
    updateTimerDisplay();
    if(timerSecs <= 0){
      clearInterval(timerInterval);
      otpResendAllowed = true;
      document.getElementById('resend-btn').classList.add('active');
    }
  }, 1000);
}
function updateTimerDisplay(){
  const mm = String(Math.floor(timerSecs/60)).padStart(2,'0');
  const ss = String(timerSecs%60).padStart(2,'0');
  const t = document.getElementById('timer');
  const t2 = document.getElementById('timer2');
  if(t) t.textContent = `${mm}:${ss}`;
  if(t2) t2.textContent = `${mm}:${ss}`;
}

function enterDashboard(){
  const name = userInfo.name || 'مستخدم';
  const first = name.split(' ')[0];
  document.getElementById('dash-name').textContent = name;
  document.getElementById('dash-contact').textContent = userInfo.contact;
  document.getElementById('welcome-name').textContent = first;
  document.getElementById('welcome-name2').textContent = first;
  const initials = name.split(' ').map(w=>w[0]).join('').toUpperCase().slice(0,2);
  document.getElementById('dash-ava').textContent = initials || 'U';
  goToPage('page-dash');
  setTimeout(()=>{ document.getElementById('sto-fill').style.width='42%'; }, 500);
}
